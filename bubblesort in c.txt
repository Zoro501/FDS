#include <stdio.h>
int main() 
{
    int m, i, j, temp,k,flag;
    printf("enter arr size");
    scanf("%d",&m);
    int arr[m];
    printf("Enter array numbers");
    for (i = 0; i < m; i++)
	{
        scanf("%d", &arr[i]);
    }
    for (i = 0; i < m-1; i++)
flag=0; {
    	printf("\nPass %d: ", i + 1);
        for (j = 0; j < m - i-1; j++) 
		{
            if (arr[j] > arr[j + 1]) 
			{
                temp = arr[j];
                arr[j] = arr[j + 1];
                arr[j + 1] = temp;
flag =1;
            }
        }
if(flag == 0)
break;
        for (k = 0; k < m; k++) {
            printf("%d ", arr[k]);
        }
        printf("\n");
    }
    printf("Sorted array in scending order is");
    for (i = 0; i < m; i++) 
	{
        printf("%d ", arr[i]);
    }
    printf("\n");
    return 0;
}
